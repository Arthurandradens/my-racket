# My Racket Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a web app that recommends tennis rackets via an adaptive quiz, with a racket comparator and SEO-optimized individual pages, monetized through affiliate links.

**Architecture:** Next.js 14+ App Router with static JSON data. Data pipeline merges Racqix API (1,147 rackets with specs) and TWU CSV (~152 rackets with performance scores) into a single `rackets.json`. Quiz runs client-side, recommendation engine scores rackets against an ideal profile generated from quiz answers.

**Tech Stack:** Next.js 14+ (App Router), TypeScript, Tailwind CSS, Vitest for testing, static JSON data (no database).

```typescript
import { writeFileSync, mkdirSync } from "fs";
import path from "path";

export interface RacqixRacquet {
  slug: string;
  brand: string;
  model: string;
  year: string | null;
  weight: number | null;
  swingweight: number | null;
  ra: number;
  balance_mm: number | null;
  head_size: number;
  string_pattern: string;
  summaries_expert_en: string;
  atp_players: string[];
  wta_players: string[];
}

export function parseRacqixResponse(response: {
  total_racquets: number;
  racquets: RacqixRacquet[];
}): RacqixRacquet[] {
  return response.racquets.map((r) => ({
    slug: r.slug,
    brand: r.brand,
    model: r.model,
    year: r.year ?? null,
    weight: r.weight ?? null,
    swingweight: r.swingweight ?? null,
    ra: r.ra,
    balance_mm: r.balance_mm ?? null,
    head_size: r.head_size,
    string_pattern: r.string_pattern,
    summaries_expert_en: r.summaries_expert_en ?? "",
    atp_players: r.atp_players ?? [],
    wta_players: r.wta_players ?? [],
  }));
}

const RACQIX_URL = "https://www.racqix.com/api/racquets?mode=minimal";

export async function fetchRacqix(): Promise<RacqixRacquet[]> {
  console.log("Fetching from Racqix API...");
  const response = await fetch(RACQIX_URL);
  if (!response.ok) {
    throw new Error(`Racqix API error: ${response.status} ${response.statusText}`);
  }
  const data = await response.json();
  const racquets = parseRacqixResponse(data);
  console.log(`Fetched ${racquets.length} racquets from Racqix.`);
  return racquets;
}

// CLI entry point
if (process.argv[1] === import.meta.filename) {
  const outDir = path.resolve("data/pipeline");
  mkdirSync(outDir, { recursive: true });
  fetchRacqix().then((racquets) => {
    const outPath = path.join(outDir, "racqix-raw.json");
    writeFileSync(outPath, JSON.stringify(racquets, null, 2));
    console.log(`Written to ${outPath}`);
  });
}
```

```typescript
import { readFileSync } from "fs";
import path from "path";

export interface TwuReview {
  name: string;
  normalized_name: string;
  scores: {
    overall: number | null;
    groundstrokes: number | null;
    volleys: number | null;
    serves: number | null;
    returns: number | null;
    power: number | null;
    control: number | null;
    maneuverability: number | null;
    stability: number | null;
    comfort: number | null;
    touch_feel: number | null;
    topspin: number | null;
    slice: number | null;
  };
  price_usd: number | null;
}

export function normalizeName(name: string): string {
  return name
    .toLowerCase()
    .replace(/\(|\)/g, "")
    .replace(/\breview\b/gi, "")
    .replace(/\bracqu[e]?t\b/gi, "")
    .replace(/\btennis\b/gi, "")
    .trim()
    .replace(/\s+/g, " ");
}

function parseScore(value: string): number | null {
  if (!value || value.trim() === "") return null;
  const num = parseFloat(value);
  if (isNaN(num)) return null;
  // Normalize 1-10 scale to 0-100
  if (num <= 10) return Math.round(num * 10);
  return num;
}

function parsePrice(value: string): number | null {
  if (!value || value.trim() === "") return null;
  const cleaned = value.replace(/[$,]/g, "");
  const num = parseFloat(cleaned);
  return isNaN(num) ? null : num;
}

interface CsvRow {
  Racquet: string;
  Overall: string;
  Groundstrokes: string;
  Volleys: string;
  Serves: string;
  Returns: string;
  Power: string;
  Control: string;
  Maneuverability: string;
  Stability: string;
  Comfort: string;
  TouchFeel: string;
  Topspin: string;
  Slice: string;
  Price: string;
}

export function parseTwuRow(row: CsvRow): TwuReview | null {
  const overall = parseScore(row.Overall);
  // Skip rows with no scores at all
  if (overall === null) return null;

  return {
    name: row.Racquet.trim(),
    normalized_name: normalizeName(row.Racquet),
    scores: {
      overall,
      groundstrokes: parseScore(row.Groundstrokes),
      volleys: parseScore(row.Volleys),
      serves: parseScore(row.Serves),
      returns: parseScore(row.Returns),
      power: parseScore(row.Power),
      control: parseScore(row.Control),
      maneuverability: parseScore(row.Maneuverability),
      stability: parseScore(row.Stability),
      comfort: parseScore(row.Comfort),
      touch_feel: parseScore(row.TouchFeel),
      topspin: parseScore(row.Topspin),
      slice: parseScore(row.Slice),
    },
    price_usd: parsePrice(row.Price),
  };
}

function parseCsvLine(line: string): string[] {
  const result: string[] = [];
  let current = "";
  let inQuotes = false;

  for (let i = 0; i < line.length; i++) {
    const char = line[i];
    if (char === '"') {
      if (inQuotes && line[i + 1] === '"') {
        current += '"';
        i++; // skip escaped quote
      } else {
        inQuotes = !inQuotes;
      }
    } else if (char === "," && !inQuotes) {
      result.push(current);
      current = "";
    } else {
      current += char;
    }
  }
  result.push(current);
  return result;
}

export function parseTwuCsv(csvContent: string): TwuReview[] {
  const lines = csvContent.trim().split("\n");
  const headers = parseCsvLine(lines[0]);
  const reviews: TwuReview[] = [];
  const seen = new Set<string>();

  for (let i = 1; i < lines.length; i++) {
    const values = parseCsvLine(lines[i]);
    const row: Record<string, string> = {};
    headers.forEach((h, idx) => {
      row[h.trim()] = (values[idx] ?? "").trim();
    });

    const review = parseTwuRow(row as CsvRow);
    if (review && !seen.has(review.normalized_name)) {
      seen.add(review.normalized_name);
      reviews.push(review);
    }
  }

  return reviews;
}
```

```typescript
import { USD_TO_BRL } from "../src/lib/constants";

const EMPTY_SCORES: RacketScores = {
  overall: null, groundstrokes: null, volleys: null, serves: null,
  returns: null, power: null, control: null, maneuverability: null,
  stability: null, comfort: null, touch_feel: null, topspin: null, slice: null,

export function fuzzyMatch(racqixName: string, twuName: string): boolean {
  // Both should already be normalized (lowercased, trimmed)
  if (racqixName === twuName) return true;
  // Check if one contains the other
  return twuName.includes(racqixName) || racqixName.includes(twuName);
}

export function assignLevels(
  weight: number | null,
  headSize: number,
  ra: number,
  balance: number | null
): string[] {
  const levels: string[] = [];
  const w = weight ?? 290; // default mid-range if unknown
  const b = balance ?? 320;

  // Iniciante: light, big head, low stiffness, head-heavy
  if (w <= 285 && headSize >= 100 && ra <= 68) {
    levels.push("iniciante");
  }
  // Intermediario: mid-range
  if (w >= 275 && w <= 310 && headSize >= 98 && headSize <= 105) {
    levels.push("intermediario");
  }
  // Avancado: heavy, small head, head-light
  if (w >= 295 && headSize <= 102) {
    levels.push("avancado");
  }

  // Fallback: if no level matched, assign intermediario
  if (levels.length === 0) {
    levels.push("intermediario");
  }

  return levels;
}

function normalizeRacqixName(r: RacqixRacquet): string {
  const parts = [r.brand, r.model, r.year].filter(Boolean);
  return parts.join(" ").toLowerCase().trim();
}

export function mergeRacketData(
  racqixData: RacqixRacquet[],
  twuData: TwuReview[]
): Racket[] {
  return racqixData.map((r) => {
    const racqixNorm = normalizeRacqixName(r);

    // Find best TWU match
    const twuMatch = twuData.find((t) => fuzzyMatch(racqixNorm, t.normalized_name));

    const priceBrl = twuMatch?.price_usd
      ? Math.round(twuMatch.price_usd * USD_TO_BRL)
      : null;

    return {
      slug: r.slug,
      brand: r.brand,
      model: `${r.model}${r.year ? ` ${r.year}` : ""}`,
      year: r.year,
      weight: r.weight,
      swingweight: r.swingweight,
      ra: r.ra,
      balance_mm: r.balance_mm,
      head_size: r.head_size,
      string_pattern: r.string_pattern,
      scores: twuMatch ? { ...twuMatch.scores } : { ...EMPTY_SCORES },
      price_brl: priceBrl,
      expert_summary_pt: r.summaries_expert_en || null, // translation is manual for now
      atp_players: r.atp_players,
      wta_players: r.wta_players,
      recommended_levels: assignLevels(r.weight, r.head_size, r.ra, r.balance_mm),
    };
  });
}
```

```typescript
import type { Racket } from "./types";

export function getAllRackets(data: Racket[]): Racket[] {
  return data;
}

export function getRacketBySlug(data: Racket[], slug: string): Racket | null {
  return data.find((r) => r.slug === slug) ?? null;
}

export function getRacketsByBrand(data: Racket[], brand: string): Racket[] {
  return data.filter((r) => r.brand.toLowerCase() === brand.toLowerCase());
}

export function searchRackets(data: Racket[], query: string): Racket[] {
  const q = query.toLowerCase();
  return data.filter(
    (r) =>
      r.model.toLowerCase().includes(q) ||
      r.brand.toLowerCase().includes(q) ||
      r.slug.includes(q)
  );
}

export function getAllBrands(data: Racket[]): string[] {
  const brands = new Set(data.map((r) => r.brand));
  return Array.from(brands).sort();
}
```

```typescript
import type { QuizAnswers, IdealProfile, SpecRange, Racket, ScoredRacket } from "./types";

export function buildIdealProfile(answers: QuizAnswers): IdealProfile {
  // Start with level defaults
  const profiles: Record<string, IdealProfile> = {
    iniciante: {
      weight: { min: 260, max: 285, importance: 0.2 },
      head_size: { min: 100, max: 110, importance: 0.3 },
      ra: { min: 0, max: 66, importance: 0.3 },
      balance_mm: { min: 325, max: 360, importance: 0.2 },
    },
    intermediario: {
      weight: { min: 280, max: 305, importance: 0.25 },
      head_size: { min: 98, max: 105, importance: 0.25 },
      ra: { min: 62, max: 70, importance: 0.25 },
      balance_mm: { min: 315, max: 330, importance: 0.25 },
    },
    avancado: {
      weight: { min: 295, max: 340, importance: 0.3 },
      head_size: { min: 95, max: 100, importance: 0.2 },
      ra: { min: 0, max: 100, importance: 0.1 },
      balance_mm: { min: 300, max: 325, importance: 0.3 },
    },
  };

  const profile = structuredClone(profiles[answers.level]);

  // Adjust for physique (beginner)
  if (answers.physique === "pequeno") {
    profile.weight.max = Math.min(profile.weight.max, 275);
    profile.weight.min = Math.max(profile.weight.min - 10, 240);
  } else if (answers.physique === "grande") {
    profile.weight.min += 10;
    profile.weight.max += 10;
  }

  // Adjust for injury
  if (answers.injury === "cotovelo" || answers.injury === "punho") {
    profile.ra.max = Math.min(profile.ra.max, 64);
    profile.ra.importance = Math.min(profile.ra.importance + 0.1, 0.4);
  } else if (answers.injury === "ombro") {
    profile.weight.max = Math.min(profile.weight.max, 285);
    profile.weight.importance = Math.min(profile.weight.importance + 0.1, 0.4);
  }

  // Adjust for improvement goals (intermediate)
  if (answers.improvement === "spin") {
    profile.head_size.min = Math.max(profile.head_size.min, 99);
    profile.head_size.importance += 0.05;
  } else if (answers.improvement === "controle") {
    profile.head_size.max = Math.min(profile.head_size.max, 100);
  } else if (answers.improvement === "potencia") {
    profile.ra.min = Math.max(profile.ra.min, 65);
    profile.head_size.min = Math.max(profile.head_size.min, 100);
  }

  // Adjust for tech preferences (advanced)
  if (answers.tech_preferences) {
    const tp = answers.tech_preferences;
    if (tp.weight === "leve") {
      profile.weight.max = 305;
      profile.weight.min = 285;
    } else if (tp.weight === "pesada") {
      profile.weight.min = 310;
      profile.weight.max = 345;
    }
    if (tp.balance === "head-light") {
      profile.balance_mm.max = 320;
    } else if (tp.balance === "head-heavy") {
      profile.balance_mm.min = 325;
    }
    if (tp.stiffness === "flexivel") {
      profile.ra.max = 63;
    } else if (tp.stiffness === "rigida") {
      profile.ra.min = 67;
    }
  }

  // Normalize importance weights to sum to 1
  const totalImportance =
    profile.weight.importance +
    profile.head_size.importance +
    profile.ra.importance +
    profile.balance_mm.importance;

  profile.weight.importance /= totalImportance;
  profile.head_size.importance /= totalImportance;
  profile.ra.importance /= totalImportance;
  profile.balance_mm.importance /= totalImportance;

  return profile;
}

function scoreSpec(value: number | null, range: SpecRange): number {
  if (value === null) return 50; // neutral score for unknown specs

  if (value >= range.min && value <= range.max) {
    return 100; // perfect fit
  }

  // Calculate distance from nearest boundary
  const distance = value < range.min ? range.min - value : value - range.max;
  const rangeSize = range.max - range.min || 1;
  const penalty = Math.min((distance / rangeSize) * 100, 100);

  return Math.max(0, 100 - penalty);
}

export function scoreRacket(racket: Racket, profile: IdealProfile): number {
  const specs: { value: number | null; range: SpecRange }[] = [
    { value: racket.weight, range: profile.weight },
    { value: racket.head_size, range: profile.head_size },
    { value: racket.ra, range: profile.ra },
    { value: racket.balance_mm, range: profile.balance_mm },
  ];

  let totalScore = 0;
  for (const { value, range } of specs) {
    totalScore += scoreSpec(value, range) * range.importance;
  }

  return Math.round(totalScore);
}

function generateReasons(racket: Racket, profile: IdealProfile, answers: QuizAnswers): string[] {
  const reasons: string[] = [];

  // Weight commentary
  if (racket.weight !== null) {
    if (racket.weight <= 280) reasons.push("Leve e facil de manobrar");
    else if (racket.weight >= 310) reasons.push("Peso solido para mais estabilidade");
  }

  // RA commentary
  if (racket.ra <= 63) reasons.push("Flexivel — mais conforto e menos impacto no braco");
  else if (racket.ra >= 70) reasons.push("Rigida — mais potencia nos golpes");

  // Head size
  if (racket.head_size >= 105) reasons.push("Area de contato generosa — mais tolerancia a erros");
  else if (racket.head_size <= 97) reasons.push("Cabeca compacta — mais precisao e controle");

  // TWU scores
  if (racket.scores.topspin && racket.scores.topspin >= 85) reasons.push("Excelente para topspin");
  if (racket.scores.comfort && racket.scores.comfort >= 85) reasons.push("Muito confortavel");
  if (racket.scores.power && racket.scores.power >= 85) reasons.push("Otima potencia");
  if (racket.scores.control && racket.scores.control >= 85) reasons.push("Controle preciso");

  // Pro players
  if (racket.atp_players.length > 0 || racket.wta_players.length > 0) {
    const players = [...racket.atp_players, ...racket.wta_players].slice(0, 2);
    reasons.push(`Usada por ${players.join(", ")}`);
  }

  return reasons.slice(0, 4); // max 4 reasons
}

export function recommend(
  rackets: Racket[],
  answers: QuizAnswers,
  topN: number = 3
): ScoredRacket[] {
  const profile = buildIdealProfile(answers);

  const scored = rackets.map((racket) => ({
    racket,
    score: scoreRacket(racket, profile),
    reasons: generateReasons(racket, profile, answers),
  }));

  scored.sort((a, b) => b.score - a.score);

  return scored.slice(0, topN);
}
```

```tsx
"use client";

import { useState, useCallback } from "react";
import { useRouter } from "next/navigation";
import { getQuizFlow } from "@/lib/quiz-config";
import type { PlayerLevel, QuizAnswers } from "@/lib/types";
import QuizStep from "@/components/QuizStep";

export default function QuizPage() {
  const router = useRouter();
  const [answers, setAnswers] = useState<Partial<QuizAnswers>>({});
  const [level, setLevel] = useState<PlayerLevel | null>(null);
  const [stepIndex, setStepIndex] = useState(0);

  const flow = level ? getQuizFlow(level) : getQuizFlow("iniciante");
  const currentQuestion = flow[stepIndex];

  const handleAnswer = useCallback(
    (questionId: string, value: string) => {
      const newAnswers = { ...answers };

      if (questionId === "level") {
        const selectedLevel = value as PlayerLevel;
        setLevel(selectedLevel);
        newAnswers.level = selectedLevel;
      } else if (questionId.startsWith("tech_preferences_")) {
        const pref = questionId.replace("tech_preferences_", "");
        newAnswers.tech_preferences = {
          weight: "media",
          balance: "equilibrada",
          stiffness: "media",
          ...newAnswers.tech_preferences,
          [pref]: value,
        };
      } else {
        (newAnswers as Record<string, string>)[questionId] = value;
      }

      setAnswers(newAnswers);

      // Check if this was the last question
      const currentFlow = newAnswers.level ? getQuizFlow(newAnswers.level as PlayerLevel) : flow;
      if (stepIndex + 1 >= currentFlow.length) {
        // Store answers and navigate to results
        sessionStorage.setItem("quizAnswers", JSON.stringify(newAnswers));
        router.push("/resultado");
      } else {
        setStepIndex(stepIndex + 1);
      }
    },
    [answers, stepIndex, flow, router]
  );

  return (
    <div className="min-h-[70vh] flex items-center justify-center px-4 py-12">
      <QuizStep
        question={currentQuestion}
        currentStep={stepIndex + 1}
        totalSteps={flow.length}
        onAnswer={handleAnswer}
      />
    </div>
  );
}
```

```tsx
  { label: "Overall", key: "overall" },

  { label: "Controle", key: "control" },

  { label: "Topspin", key: "topspin" },
  { label: "Manobrabilidade", key: "maneuverability" },

];

  const nums = values.map((v, i) => (v !== null ? { v, i } : null)).filter(Boolean) as { v: number; i: number }[];
  if (nums.length < 2) return null;
  nums.sort((a, b) => (higherIsBetter ? b.v - a.v : a.v - b.v));
  return nums[0].i;
}
```

```tsx
"use client";

import { useState, useEffect, useMemo } from "react";
import { useSearchParams, useRouter } from "next/navigation";
import type { Racket } from "@/lib/types";
import { searchRackets, getRacketBySlug } from "@/lib/rackets";
import racketData from "@/data/rackets.json";
import CompareTable from "@/components/CompareTable";

export default function CompararPage() {
  const searchParams = useSearchParams();
  const router = useRouter();
  const [selectedSlugs, setSelectedSlugs] = useState<string[]>([]);
  const [searchQuery, setSearchQuery] = useState("");

  // Load from URL params on mount
  useEffect(() => {
    const r = searchParams.get("r");
    if (r) {
      setSelectedSlugs(r.split(",").filter(Boolean));
    }
  }, [searchParams]);

  const selectedRackets = useMemo(
    () =>
      selectedSlugs
        .map((slug) => getRacketBySlug(racketData as Racket[], slug))
        .filter(Boolean) as Racket[],
    [selectedSlugs]
  );

  const searchResults = useMemo(() => {
    if (searchQuery.length < 2) return [];
    return searchRackets(racketData as Racket[], searchQuery)
      .filter((r) => !selectedSlugs.includes(r.slug))
      .slice(0, 8);
  }, [searchQuery, selectedSlugs]);

  const addRacket = (slug: string) => {
    if (selectedSlugs.length >= 3) return;
    const newSlugs = [...selectedSlugs, slug];
    setSelectedSlugs(newSlugs);
    setSearchQuery("");
    router.replace(`/comparar?r=${newSlugs.join(",")}`);
  };

  const removeRacket = (slug: string) => {
    const newSlugs = selectedSlugs.filter((s) => s !== slug);
    setSelectedSlugs(newSlugs);
    router.replace(newSlugs.length > 0 ? `/comparar?r=${newSlugs.join(",")}` : "/comparar");
  };

  return (
    <div className="max-w-5xl mx-auto px-4 py-12">
      <h1 className="text-3xl font-bold mb-2">Comparar raquetes</h1>
      <p className="text-gray-600 mb-8">
        Selecione ate 3 raquetes para comparar lado a lado.
      </p>

      {/* Search to add */}
      {selectedSlugs.length < 3 && (
        <div className="mb-8 relative">
          <input
            type="text"
            placeholder="Buscar raquete para adicionar..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="w-full md:w-96 border-2 border-gray-200 rounded-lg px-4 py-3 focus:border-green-600 focus:outline-none"
          />
          {searchResults.length > 0 && (
            <div className="absolute z-10 mt-1 w-full md:w-96 bg-white border border-gray-200 rounded-lg shadow-lg max-h-64 overflow-y-auto">
              {searchResults.map((r) => (
                <button
                  key={r.slug}
                  onClick={() => addRacket(r.slug)}
                  className="w-full text-left px-4 py-3 hover:bg-gray-50 border-b border-gray-100 last:border-0"
                >
                  <span className="text-sm text-gray-500">{r.brand}</span>{" "}
                  <span className="font-medium">{r.model}</span>
                </button>
              ))}
            </div>
          )}
        </div>
      )}

      {/* Selected rackets chips */}
      {selectedSlugs.length > 0 && (
        <div className="flex flex-wrap gap-2 mb-6">
          {selectedRackets.map((r) => (
            <span
              key={r.slug}
              className="inline-flex items-center gap-1 bg-green-100 text-green-800 px-3 py-1.5 rounded-full text-sm font-medium"
            >
              {r.brand} {r.model}
              <button
                onClick={() => removeRacket(r.slug)}
                className="ml-1 text-green-600 hover:text-green-900"
              >
                &#10005;
              </button>
            </span>
          ))}
        </div>
      )}

      {/* Comparison table */}
      {selectedRackets.length > 0 ? (
        <CompareTable rackets={selectedRackets} />
      ) : (
        <div className="text-center py-20 text-gray-400">
          <p className="text-lg">Nenhuma raquete selecionada.</p>
          <p className="text-sm">Use a busca acima para adicionar raquetes.</p>
        </div>
      )}
    </div>
  );
}
```

```tsx
"use client";

import { useState, useMemo } from "react";
import Link from "next/link";
```
